    function calculateResult() {

        const form = document.getElementById('quiz-form');

        const totalQuestions = 15; // Definindo o número total de perguntas para o cálculo

        let score = 0;


        for (let i = 1; i <= totalQuestions; i++) {

            const answer = form['q' + i].value;

            if (answer) score += parseInt(answer);

        }


        const result = document.getElementById('result');


        if (score >= 30 && score <= 59) {

            result.textContent = `Sua pontuação é ${score}. Nível 1 - Baixo. O uso de pornografia está em um nível controlado.`;

        } else if (score >= 60 && score <= 89) {

            result.textContent = `Sua pontuação é ${score}. Nível 2 - Moderado. Sinais de consumo que podem interferir em algumas áreas.`;

        } else if (score >= 90 && score <= 119) {

            result.textContent = `Sua pontuação é ${score}. Nível 3 - Alto. O consumo afeta várias áreas da vida.`;

        } else if (score >= 120 && score <= 139) {

            result.textContent = `Sua pontuação é ${score}. Nível 4 - Crítico. A pornografia tem impacto significativo na vida pessoal e social.`;

        } else if (score >= 140 && score <= 150) {

            result.textContent = `Sua pontuação é ${score}. Nível 5 - Severamente Crítico. Intervenção imediata é necessária.`;

        } else {

            result.textContent = 'Por favor, responda todas as perguntas.';

        }

    }
